// 백준 13975 : 파일 합치기 3
#include <vector>
#include <string>
#include <iostream>
#include <queue>

using namespace std;

int main() {
	// 입력 받기
	int T;
	cin >> T;
	vector<vector<int>> files(T);
	for (int i = 0; i < T; i++) {
		int N;
		cin >> N;
		for (int j = 0; j < N; j++) {
			int size;
			cin >> size;
			files[i].push_back(size);
		}
	}
	
	// 그리디
	vector<long long> answer;
	for(int i=0; i<T; i++){
		priority_queue<long long, vector<long long>, greater<long long>> pq;
		for (int j = 0; j < files[i].size(); j++) {
			pq.push(files[i][j]);
		}

		long long total = 0;
		while (1) {
			long long a = pq.top();
			pq.pop();
			long long b = pq.top();
			pq.pop();

			long long cost = a + b; // 두 파일을 합치는 비용
			total += cost; // total 계산

			pq.push(cost);

			if (pq.size() == 1) {
				answer.push_back(total);
				break;
			}
		}
	}

	for (long long ans : answer) {
		cout << ans << '\n';
	}
}
