// 백준 7983 : 내일 할거야
#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

bool cmp(pair<int, int> a, pair<int, int> b) {
	return a.second>b.second;
}

int main() {
	int n; // 과제의 개수 n
	vector<pair<int, int>> assignment; // 과제의 정보를 저장 assignment

	cin >> n;
	for (int i = 0; i<n; i++) {
		int d, t; // 각 과제의 소요일 d, 마감일 t
		cin >> d >> t;
		assignment.push_back(make_pair(d, t));
	}

	sort(assignment.begin(), assignment.end(), cmp); // 과제의 마감일(t)를 기준으로 내림차순 정렬

	int maxDayOff = assignment.front().second - assignment.front().first; 
	for (int i = 1; i<n; i++) {
		if (maxDayOff >= assignment[i].second) {
			maxDayOff = assignment[i].second - assignment[i].first;
		}
		else {
			maxDayOff = maxDayOff - assignment[i].first;
		}
	}
	cout << maxDayOff << endl;

	return 0;
}
