// 백준 3167 : 기차표 검사
#include <iostream>
#include <vector>
using namespace std;

int main() {
	int n, k; // 기차가 지나는 역의 개수 n, 직원이 표를 검사할 역의 간격 k
	cin >> n >> k;

	vector<pair<int, int>> station; // 각 역에서의 승하차할 승객의 수를 저장할 벡터 station
	for (int i = 0; i < n; i++) {
		int x, y; // 하차할 승객과 승차할 승객의 수
		cin >> x >> y;
		station.push_back({x,y});
	}

	for (int i = 0; i < 2; i++) {
		int checked = 0; // 표 검사를 받은 승객의 수
		int notChecked = 0; // 표 검사를 받지 않은 승객의 수
		int answer = 0; // 출력값

		for (int j = 0; j < n; j++) {
			if (j == 1 || (j - 1) % k == 0) { // 모든 승객의 기차표를 검사
				checked += notChecked;
				notChecked = 0;
			}

			// 승객 하차 (표를 검사한 승객이 최대한 많이 내리도록 함)
			int tmp = station[j].first;
			if (i == 0) { // 최솟값을 구하는 경우
				if (tmp > checked) {
					tmp -= checked;
					checked = 0;
					notChecked -= tmp;
					answer += tmp; // 하차한 승객중 한번도 표를 검사받지 않은 승객의 수
				}
				else {
					checked -= tmp;
				}
			}
			else { // 최댓값을 구하는 경우
				if (tmp > notChecked) {
					tmp -= notChecked;
					answer += notChecked; // 하차한 승객중 한번도 표를 검사받지 않은 승객의 수
					notChecked = 0;
					checked -= tmp;
				}
				else {
					notChecked -= tmp;
					answer += tmp;
				}
			}

			// 승객 승차
			notChecked += station[j].second;
		}
		cout << answer << ' ';
	}
	cout << '\n';
}
