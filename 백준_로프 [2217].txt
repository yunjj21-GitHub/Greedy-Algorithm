// 백준 2217 : 로프
#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

bool compare(int a, int b) {
	return a > b;
}

int main() {
	int n; // 입력될 로프의 개수
	cin >> n;
	int *p = new int[n]; // 각 로프들이 버틸 수 있는 최대 중량
	for (int i = 0; i < n; i++) {
		cin >> p[i];
	}

	// 내림차순 정렬
	sort(p, p + n, compare);

	int max = p[0];
	for (int i = 1; i < n; i++) {
		if (p[i]*(i+1) > max) max=p[i]*(i + 1);
	}
	cout << max << endl;
}
