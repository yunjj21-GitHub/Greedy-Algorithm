// 백준 1202 : 보석 도둑
#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
#include <queue>
using namespace std;

bool cmp(pair<int, int>a, pair<int, int>b) {
	return a.first < b.first;
}
bool cmpAry(int a, int b) {
	return a < b;
}

int main() {
	int n, k; // 보석의 개수 n, 가방의 개수 k
	cin >> n >> k;
	int m, v; // 각 보석의 무게 m, 가격 v
	vector<pair<int, int>> jewelry; // 각 보석에 대한 정보를 저장
	for (int i = 0; i < n; i++) {
		cin >> m >> v;
		jewelry.push_back(make_pair(m, v));
	}
	sort(jewelry.begin(), jewelry.end(), cmp); // 보석의 무게가 낮은 순(오름차순)으로 정렬

	int *bag = new int[k]; // 각 가방에 대한 정보를 저장
	for (int i = 0; i < k; i++) {
		cin >> bag[i]; // 각 가방에 담을 수 있는 최대 무게
	}
	sort(bag, bag + k, cmpAry); // 각 가방에 담을 수 있는 최대 무게가 낮은 순(오름차순)으로 정렬

	priority_queue<int> pq;
	int j = 0;
	long long maxPrice = 0;
	
	for (int i = 0; i < k; i++) {

		while (j<n && jewelry[j].first <= bag[i]) {
			pq.push(jewelry[j].second);
			j++;
		}

		if (!pq.empty()) {
			maxPrice += pq.top();
			pq.pop();
		}
	}
	cout << maxPrice << endl;
}
