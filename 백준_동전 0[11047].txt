// 백준 11047 : 동전 0
#include <iostream>
#include <string>
using namespace std;

int main() {
	int n, k; // n이 동전 종류의 개수, k원(가치)를 만들 것임
	int counter = 0; // 사용된 동전의 개수를 계산할 변수
	cin >> n >> k;
	int *p = new int[n]; // 각 동전의 가치를 저장할 배열
	for (int i = 0; i < n; i++) {
		cin >> p[i];
	}

	// 필요한 동전 개수 계산
	for (int i = n - 1; i >= 0; i--) { // 동전의 큰 수 부터 가치 k와 비교
		while (k>=p[i]) {
			k -= p[i];
			counter++;
			if (k < p[i]) break;
		}
	}
	cout << counter << endl;
}
