// 백준 1715 : 카드 정렬하기
#include <vector>
#include <string>
#include <iostream>
#include <queue>

using namespace std;

int n;
priority_queue<int, vector<int>, greater<int>> cards; // 자동 오름차순 정렬
int ans = 0;

int main() {
	cin >> n;
	for (int i = 0; i < n; i++) {
		int size;
		cin >> size;
		cards.push(size);
	}

	while (cards.size() > 1) {
		int x = cards.top();
		cards.pop();
		int y = cards.top();
		cards.pop();

		ans += (x + y);
		cards.push(x + y);
	}

	cout << ans << '\n';
}
