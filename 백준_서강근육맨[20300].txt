#include <vector>
#include <string>
#include <algorithm>
#include <iostream>

using namespace std;

int main() { 
	int N; // 운동기구의 개수
	cin >> N;
	vector<long long> equip(N); // 각 운동기구의 근손실량을 저장
	for (int i = 0; i < N; i++) {
		cin >> equip[i];
	}
	sort(equip.begin(), equip.end()); // 근손실량을 오름차순 정렬

	long long answer = 0;
	if (N % 2 != 0) { // N이 홀수라면
		answer = equip[N - 1]; // 근손실이 가장 큰 운동기구를 마지막날 사용한다.
		N--;
	}
	
	for (int i = 0; i < N / 2; i++) {
		long long M = equip[i] + equip[(N - 1) - i];
		if (M > answer) answer = M;
	}
	cout << answer << '\n';
}
