// 백준 1781 : 컵라면
#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
#include <queue>
using namespace std;

bool compare(pair<int, int>a, pair<int, int>b) {
	return a.first > b.first; // 내림차순
}

int main() {
	int n; // 숙제의 개수
	cin >> n;
	vector<pair<int, int>> problem;
	int de, cup; // 데드라인과 컵라면 수
	for (int i = 0; i < n; i++) {
		cin >> de >> cup;
		problem.push_back(make_pair(de, cup));
	}
	sort(problem.begin(), problem.end(), compare);

	int maxCup = 0; // 최대 컵라면 수
	
	priority_queue<int> pq;
	int index = 0;
	for (int t = n; t > 0; t--) { // 시간을 n부터 1까지 흐르게 함
		for (int i = index; i < n; i++) { // t에 해당하는 데드라인을 힙에 넣는다.
			if (problem[i].first != t) {
				index = i;
				break;
			}
			pq.push(problem[i].second);
		}
		if (!pq.empty()) {
			maxCup += pq.top();
			pq.pop();
		}
	}

	// 출력
	cout << maxCup << endl;
}
