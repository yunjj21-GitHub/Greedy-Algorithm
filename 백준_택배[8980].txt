#include <vector>
#include <string>
#include <iostream>
#include <algorithm>

using namespace std;

int N, C; // 마을의 개수 n, 트럭의 용량 c
int M; // 박스 정보의 개수 m
vector<vector<int>> box; // 박스 정보를 저장
int truck[2001]; // 각 마을에서의 트럭에 실린 박스 수를 저장
int ans;

bool cmp(vector<int>a, vector<int> b) {
	// 받는 마을 번호를 기준으로 오름차순
	if (a[1] == b[1]) { // 받는 마을 번호가 동일하다면 보내는 마을 번호를 기준으로 오름차순
		return a[0] < b[0];
	}
	else {
		return a[1] < b[1];
	}
}

int main() {
	cin >> N >> C;
	cin >> M;
	for (int i = 0; i < M; i++) {
		int x, y, num;
		cin >> x >> y >> num;
		box.push_back({ x, y, num });
	}

	sort(box.begin(), box.end(), cmp);

	for (int i = 0; i < M; i++) {
		int empty = box[i][2];
		for (int j = box[i][0]; j < box[i][1]; j++) {
			if (C - truck[j] < empty) empty = C - truck[j];
		}

		// 갱신
		for (int j = box[i][0]; j < box[i][1]; j++) {
			truck[j] += empty;
		}
		ans += empty;
	}

	cout << ans << '\n'; // 답안 출력
}
