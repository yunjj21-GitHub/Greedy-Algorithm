// 백준 1464 : 뒤집기 3
#include <iostream>
#include <string>
using namespace std;

int main() {
	string s;
	cin >> s;
	for (int i = 1; i < s.size(); i++) { // 1부터 N까지의 수를 차례로 생각
		if (s.front() >= s[i]) { // 조건을 만족한다면 뒤집는다.
			for (int j = 0; j <= (i - 1) / 2; j++) { // 0~i-1까지를 뒤집는다.
				char tmp = s[j];
				s[j] = s[(i - 1)- j];
				s[(i - 1)- j] = tmp;
			}

			for (int j = 0; j <= i / 2; j++) { // 0~i까지를 뒤집는다.
				char tmp = s[j];
				s[j] = s[i - j];
				s[i - j] = tmp;
			}
		}
	}
	cout << s << endl;
}
