#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

bool compare(double a, double b){
	return a > b;
}

int main() {
	int n, m, k; // 예선참가자 n, 장르의 종류 m, 본선 진출자 k
	cin >> n >> m >> k;
	
	double con[100] = { 0.0, }; // 참가자의 최대능력치를 저장
	int par; // 각 참가자
	double ability; // 각 참가자의 능력치
	for (int i = 0; i < m; i++) {
		for (int j = 0; j < n; j++) {
			cin >> par;
			cin >> ability;
			if (con[par] < ability) con[par] = ability;
		}
	}
	sort(con, con + (n+1), compare); // par이 1부터 n까지 이기에

	double max = 0.0;
	for (int i = 0; i < k; i++) {
		max += con[i];
	}
	cout << fixed;
	cout.precision(1);
	cout << max << endl;
}
