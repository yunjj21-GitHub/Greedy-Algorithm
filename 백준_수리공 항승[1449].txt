// 백준 1449 : 수리공 항승
#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

bool compare(int a, int b) {
	return a < b; // 오름차순
}

int main() {
	int n, l; // 물이 새는 곳의 개수 n, 테이프의 길이 l
	cin >> n >> l;
	int *x = new int[n]; // 물이 새는 곳의 위치를 저장할 배열 x
	for (int i = 0; i < n; i++) {
		cin >> x[i];
	}
	// 입력되는 물이 새는 곳의 위치가 가장 왼쪽에서 가까운 순서대로 입력된게 아니라는 가정
	sort(x, x + n, compare);

	double tail = 0; // 현 새는 위치의 왼쪽에 0.5를 여유롭게 붙이고 오른쪽에 테이프 끝의 위치
	int minTape = 0; // 필요한 테이프의 최소 개수

	for (int i = 0; i < n; i++) {
		if (x[i] + 0.5 <= tail) {
			continue;
		}
		else {
			tail = x[i] + l - 0.5;
			minTape++;
		}
	}
	cout << minTape << endl;
}
