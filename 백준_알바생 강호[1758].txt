#include <vector>
#include <string>
#include <algorithm>
#include <iostream>

using namespace std;

bool cmp(int x, int y) {
	return x > y;
}

int main() {
	int N;
	cin >> N;
	vector<int> tips(N);
	for (int i = 0; i < N; i++) {
		cin >> tips[i];
	}

	long long answer = 0;
	sort(tips.begin(), tips.end(), cmp); // 팁의 액수에 따라 내림차순 정렬

	for (int i = 0; i < N; i++) { // 등수는 i+1
		if (tips[i] - i >= 0) answer += (long long)(tips[i] - i);
	}
	cout << answer << '\n';
}
