// 백준 2012 : 등수 매기기
#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

bool compare(int a, int b) {
	return a < b; //오름차순
}

int main() {
	int n;
	cin >> n;
	int *expRank = new int[n];
	long long minComplaint = 0;
	for (int i = 0; i < n; i++) {
		cin >> expRank[i];
	}
	sort(expRank, expRank + n, compare);

	for (int i = 0; i < n; i++) { // 단순 반복 횟수 n회
		minComplaint += abs(expRank[i] - (i+1)); // 예상 등수 expRank[i], 실제 등수 i+1
	}
	cout << minComplaint << endl;
}
