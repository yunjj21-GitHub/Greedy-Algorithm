#include <vector>
#include <string>
#include <iostream>

using namespace std;

int main() {
	string board;
	cin >> board;

	string answer = "";
	int cnt = 0;
	for (int i = 0; i < board.size(); i++) {
		if (board[i] == 'X') cnt++;
		else if (board[i] == '.') {
			if (cnt == 2) {
				answer += "BB";
			}
			cnt = 0;
			answer += '.';
		}

		if (cnt == 4) {
			answer += "AAAA";
			cnt = 0;
		}
	}
	if (cnt == 2) answer += "BB";

	if (answer.size() != board.size()) cout << -1 << '\n';
	else cout << answer << '\n';
}
